    package ui;
/**
 *
 * @author Vladimir
 */
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.*;
import java.util.logging.Level;
import java.util.logging.Logger;
import javax.swing.*;

public class Window {   
    public JTextArea mainArea;
    public static void main(String[] args) {
        JFrame frame = new Frame_1();
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}

class Frame_1 extends JFrame {
    public JTextArea mainArea;
    public Frame_1(){
        setTitle("Window");
        setSize(600, 700);
        setLayout( new BorderLayout() );
        
        JMenuBar menuBar = new JMenuBar();
        setJMenuBar(menuBar);
        
        final JFileChooser chooser = new JFileChooser();
        chooser.setCurrentDirectory(new File("."));
        
        JMenu menu = new JMenu("File");
        menuBar.add(menu);
        
        JMenuItem openItem = new JMenuItem("Open");
        menu.add(openItem);
        openItem.addActionListener(new ActionListener(){
            public void actionPerformed(ActionEvent event){
                int result = chooser.showOpenDialog(null);
                String selectedFileName = mainArea.getText().trim(); 
                    if(selectedFileName.length() > 0) 
                    { 
                    try {
                        readFile(selectedFileName);
                    } catch (FileNotFoundException ex) {
                        Logger.getLogger(Frame_1.class.getName()).log(Level.SEVERE, null, ex);
                    } catch (IOException ex) {
                        Logger.getLogger(Frame_1.class.getName()).log(Level.SEVERE, null, ex);
                    }
                    }
            }
        });
        
        JMenuItem exitItem = new JMenuItem("Exit");
        menu.add(exitItem);
        exitItem.addActionListener(new ActionListener(){
            public void actionPerformed(ActionEvent event) {
                System.exit(0);
            }
        });
        
        framePanel panel = new framePanel();
        add(panel);
    }
    private void readFile(String selectedFileName) throws FileNotFoundException, IOException 
    { 
            FileReader fr = new FileReader(selectedFileName); 
            StringBuffer sb = new StringBuffer(); 
            int symbol; 
            while((symbol = fr.read()) != -1) 
            { 
                sb.append((char)symbol); 
            } 
            mainArea.setText(sb.toString()); 
    }
}
class framePanel extends JPanel {    
    public framePanel(){
        JTextArea mainArea = new JTextArea("", 50, 50);
        add(mainArea);
    }
}
